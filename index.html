            function styles() {
              return src('./scss/style.scss')
                .pipe(plumber())
                .pipe(sourcemaps.init())
                .pipe(sass())
                .pipe(postcss([
                  autoprefixer({browsers: ['last 2 version']}),
                ]))
                .pipe(sourcemaps.write('/'))
                .pipe(dest('./css/'))
                .pipe(browserSync.stream());
            }
            exports.styles = styles;
